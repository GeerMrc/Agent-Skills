```python
def check_reference_depth():
```

```bash
import re
from pathlib import Path

VERSION_PATTERN = r'v0\.1\.1\.1'

def check_version_consistency():
    mismatches = []
    for md_file in Path('.').rglob('*.md'):
        content = md_file.read_text()
        # 检查版本号
        if re.search(VERSION_PATTERN, content):
            print(f"✅ {md_file}: 版本号一致")
        else:
            mismatches.append(md_file)
    return mismatches

if __name__ == '__main__':
    mismatches = check_version_consistency()
    if len(mismatches) == 0:
        print("✅ 所有文档版本号一致")
    else:
        print(f"⚠️ {len(mismatches)}个文档版本号不一致")
```

### B. 自动化检查脚本

#### check-all.py

```python
#!/usr/bin/env python3
"""
完整的自动化检查脚本

功能：
- 检查SKILL.md行数
- 检查引用深度
- 检查版本号一致性
- 检查Python语法
- 检查PEP8规范
- 检查测试覆盖率
"""

import os
import sys
import subprocess
from pathlib import Path

def check_skill_lines():
    """检查SKILL.md行数"""
    result = subprocess.run(['wc', '-l', 'SKILL.md'], capture_output=True, text=True)
    lines = int(result.stdout.split()[0])
    if lines <= 200:
        print(f"✅ SKILL.md行数: {lines}行（≤200行）")
        return True
    else:
        print(f"❌ SKILL.md行数: {lines}行（>200行）")
        return False

def check_reference_depth():
    """检查引用深度"""
    max_depth = 0
    for md_file in Path('references').rglob('*.md'):
        depth = md_file.relative_to('references').parts.count('.')
        max_depth = max(max_depth, depth)
    if max_depth <= 1:
        print(f"✅ 引用深度: {max_depth}层（≤1层）")
        return True
    else:
        print(f"❌ 引用深度: {max_depth}层（>1层）")
        return False

def check_python_syntax():
    """检查Python语法"""
    result = subprocess.run(
        ['python', '-m', 'py_compile', 'scripts/check-tokens.py'],
        capture_output=True
    )
    if result.returncode == 0:
        print("✅ Python语法检查通过")
        return True
    else:
        print("❌ Python语法检查失败")
        return False

def main():
    """主函数"""
    print("开始自动化检查...\n")
    results = []
    results.append(check_skill_lines())
    results.append(check_reference_depth())
    results.append(check_python_syntax())
    print(f"\n检查完成: {sum(results)}/{len(results)}项通过")
    return all(results)

if __name__ == '__main__':
    success = main()
    sys.exit(0 if success else 1)
```
